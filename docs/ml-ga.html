def ll (b: (Fin 8)=>Float) : Float = neg $ sum (log (map (\ x. (exp x) + 1) ((map (\ yi. 1 - 2*yi) y)*(x **. b))))

def one_step (learning_rate: Float) (b0: (Fin 8)=>Float) : (Fin 8)=>Float = b0 + learning_rate .* gll b0
